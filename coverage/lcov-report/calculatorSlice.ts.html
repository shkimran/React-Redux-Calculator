const compute = (exp: string[]): number | string => {
  const toNum = (s: string) => parseFloat(s);
  const prec: Record<string, number> = { '+': 1, '−': 1, '×': 2, '÷': 2 };
  const output: (string | number)[] = [];
  const ops: string[] = [];
 
  for (const token of exp) {
    if (!isOperator(token)) {
      output.push(token);
    } else {
      while (ops.length && prec[ops[ops.length - 1]] >= prec[token]) {
        output.push(ops.pop()!);
      }
      ops.push(token);
    }
  }
  while (ops.length) output.push(ops.pop()!);
 
  const stack: number[] = [];
  for (const token of output) {
    if (typeof token === 'string' && isOperator(token)) {
      const b = stack.pop()!;
      const a = stack.pop()!;
      let result = 0;
      switch (token) {
        case '+': result = a + b; break;
        case '−': result = a - b; break;
        case '×': result = a * b; break;
        case '÷':
          if (b === 0) {
            toast.error("Cannot divide by zero");
            return 'Error';

          result = a / b;
          break;
      }
      stack.push(Number(result.toPrecision(12)));
    } else {
      stack.push(toNum(token as string));
    }
  }
  return stack[0];
};
